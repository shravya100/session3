	{
		System.out.println( "Enter how much money you want to deposit: ");
		double amount = sc.nextDouble();
		if(amount<0)
		{
			throw new BadFormatException("negative amount is invalid");

public double withdrawl() throws InsufficientBalance
{
	System.out.println("Enter how much money you want to withdraw: ");
	double amount = sc.nextDouble();
	
	if(amount<0)
	{
		throw new BadFormatException("negative amount is invalid");
	}

	if(amount>accountBalance)
	{
		throw new InsufficientBalance("Available balance:Rs "+accountBalance+" WithDrawl request :"+amount+" transaction cannot completed");
		
	}
	 
	 accountBalance-=amount;
     System.out.println("\n Available balance: " + accountBalance); 
	
	 return accountBalance;
}
